# 13章
- 配列
- 要素数が必要
```c
#include <stdio.h>

int main(void){
    //残りには0が代入される
    //要素数を省略すると{}内の個数が要素数となる
    int array[10] = {100, 67, 45};
    array[2] = 100;
    printf("%d\n", array[2]);
    array[2] ++;
    printf("%d\n", array[2]);
    printf("%d\n", array[9]);
    printf("------------\n");

    //配列に対してループを回す
    int length = sizeof(array) / sizeof(array[0]);
    for (int i=0; i<length; i++){
        printf("%d\n", array[i]);
    }
    return 0;
}
```
- memcpyで配列をコピー
- copy元がcopy先より長いとはみ出してバッファオーバーフローするので注意
```c
#include <stdio.h>
#include <string.h>

int main(void){

    //memcpy(to_copy, from_copy, 配列全体のsize)

    int array1[5];
    int array2[5];
    
    for (int i=0; i < sizeof(array1) / 4; i++){
        array1[i] = i;
    }

    memcpy(array2, array1, sizeof(array1));

    for (int i=0; i < sizeof(array2) / 4; i++){
        printf("%d\n", array2[i]);
    }

    return 0;
}
```
